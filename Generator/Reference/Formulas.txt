// No a; end speed from start speed, displacement and time.
v=u+s/t

// No s; acceleration from start speed, end speed and time.
a=(v-u)/t

// No t; end speed from start speed, acceleration and displacement.
v^2=u^2+2*a*s

// No v; displacement from start speed, acceleration and time.
s=u*t+0.5*a*t^2

// No u; displacement from end speed, acceleration and time.
s=v*t-a*t^2+0.5*a*t^2
